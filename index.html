    <script>

        function savePreferences() {

            const input = document.getElementById('preferences-input').value;

            if (!input) {

                alert('Please enter your preferences.');

                return;

            }

            localStorage.setItem('preferences', input);

            alert('Preferences saved! Content will be curated based on your input.');

            displayContent();

        }


        function displayContent() {

            const contentList = document.getElementById('content-list');

            const preferences = localStorage.getItem('preferences');

            if (!preferences) {

                contentList.innerHTML = '<p>No content yet. Start by adding preferences above!</p>';

                return;

            }


            // Simulated curated content based on preferences

            const curatedContent = [

                { title: 'Interesting Article on ' + preferences, link: 'https://example.com/article1' },

                { title: 'Social Media Post about ' + preferences, link: 'https://example.com/post2' },

                { title: 'Resource on ' + preferences, link: 'https://example.com/resource3' }

            ];


            contentList.innerHTML = '';

            curatedContent.forEach(item => {

                const div = document.createElement('div');

                div.classList.add('content-item');

                div.innerHTML = `<h3>${item.title}</h3><a href="${item.link}" target="_blank">Read More</a>`;

                contentList.appendChild(div);

            });

        }


        // Load content on page load

        document.addEventListener('DOMContentLoaded', displayContent);

    </script>
